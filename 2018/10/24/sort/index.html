//冒泡排序
public static void BubbleSort(int[] array){
    for(int i=0;i<array.length-1;i++){
        boolean flag=false;//设置是否发生交换的标志
        for(int j=0;j<array.length-i-1;j++){
            if(array[j]>array[j+1])
            {//每轮找到一个最大的数放在右边
                int t=array[j];
                array[j]=array[j+1];
                array[j+1]=t;
                flag=true;//发生交换
            }            
        }
        if(flag==false) break;//这轮循环无交换，说明排序完成，退出循环
    }
}
//简单选择排序
public static void SelectSort(int[] array){
    for(int i=0;i<array.length-1;i++){
        int k=i;
        for(int j=i;j<array.length;j++){
            if(array[k]>array[j])
                k=j;//选择当前最小的数
        }
        int t=array[i];
        array[i]=array[k];
        array[k]=t;
    }
}
//直接插入排序
public static void InsertSort(int[] array){
    for(int i=1;i<array.length;i++){
        int temp=array[i];int j;
        for(j=i;j>=1&&temp<array[j-1];j--)//把大于temp的数往右移
            array[j]=array[j-1];
        array[j]=temp;
    }
}
//希尔排序
public static void ShellSort(int[] array){
    //定义一个增长序列，即分割数组的增量
    for(int gap=array.length/2;gap>0;gap=gap/2){
        for(int i=gap;i<array.length;i++){
            int temp=array[i];int j;
            for(j=i;j>=gap&&temp<array[j-gap];j=j-gap)
                array[j]=array[j-gap];
            array[j]=temp;
        }
    }        
}
//快速排序
public static int partition(int[] array,int low,int high){
    int temp=array[low];//选取第一个值作为枢轴
    while(low<high){
        //将比枢轴小的交换到低端
        while(low<high&&array[high]>=temp)--high;
        array[low]=array[high];//采用替换而不是交换的方式
        //将比枢轴大的交换到高端
        while(low<high&&array[low]<=temp)++low;
        array[high]=array[low];
    }
    //枢轴所在位置赋值
    array[low]=temp;
    return low;
}
public static void QuickSort(int[] array,int low,int high){
    int loc;
    if(low<high){
        loc=partition(array,low,high);//将数组一分为二
        QuickSort(array,low,loc-1);//对左半部分进行递归排序
        QuickSort(array,loc+1,high);//对右半部分进行递归排序
    }
}
//堆排序
public static void Heapadjust(int[] array,int s,int m){
    int temp=array[s];
    for(int j=2*s;j<=m;j=j*2){
        if(j<m&&array[j]<array[j+1])j++;//j<m很重要，左右孩子选择最大者
        if(array[j]<=temp)break;
        array[s]=array[j];
        s=j;
    }
    array[s]=temp;
}
public static void HeapSort(int[] array){
    int i;
    for(i=array.length/2-1;i>=0;i--)
        Heapadjust(array,i,array.length-1);
    for(i=array.length-1;i>0;i--){
        int temp=array[i];
        array[i]=array[0];
        array[0]=temp;
        Heapadjust(array,0,i-1);
    }
}
//归并排序(递归)
public static void Merge(int[] array,int[] tmp,int s,int m,int n){
    int la,lb,lc;
    la=s;lb=m+1;lc=s;
    int num=n-s+1;
    while(la<=m&&lb<=n){
        if(array[la]<array[lb])tmp[lc++]=array[la++];
        else
            tmp[lc++]=array[lb++];
    }
    while(la<=m)tmp[lc++]=array[la++];
    while(lb<=n)tmp[lc++]=array[lb++];
    for(int i=0;i<num;i++,n--)
        array[n]=tmp[n];
}
public static void MergeSort1(int[] array,int[] tmp,int s,int n){
    int m;
    if(s<n){
        m=(s+n)/2;
        MergeSort1(array,tmp,s,m);
        MergeSort1(array,tmp,m+1,n);
        Merge(array,tmp,s,m,n);
    }
}
//归并排序(迭代)
public static void MergePass(int[] array,int[] tmp,int gap,int n){
    int i=0;
    //归并gap长度的两个相邻子表
    for(i=0;i<n-2*gap+1;i=i+2*gap)
        Merge(array,tmp,i,i+gap-1,i+2*gap-1);
    //余下两个子表
    if(i<n-gap+1)
        Merge(array,tmp,i,i+gap-1,n);
    //余下一个子表
    else
        for(int j=i;j<=n;j++)
            tmp[j]=array[j];
}

public static void MergeSort2(int[] array){
    int tmp[]=new int[array.length];
    int k=1;
    while(k<array.length){
        MergePass(array,tmp,k,array.length-1);
        k=2*k;
        MergePass(tmp,array,k,array.length-1);
        k=2*k;            
    }
}
